 geom_tile() +

 scale_fill_identity() +
 scale_y_continuous(breaks=NULL) +
 theme_void(base_family="Roboto Condensed") +
 coord_flip(ylim=c(1,12)) +
 theme(axis.text = element_text(color = "black", family="Roboto Condensed", hjust=1)) +
